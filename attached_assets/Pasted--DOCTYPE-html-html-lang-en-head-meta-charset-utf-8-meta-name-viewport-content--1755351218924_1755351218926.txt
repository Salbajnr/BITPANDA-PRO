    gradient.addColorStop(0, type === 'line' ? 'rgba(212, 175, 55, 0.3)' : 'rgba(16, 185, 129, 0.3)');
    gradient.addColorStop(1, 'rgba(14, 17, 22, 0)');

      datasets: [{

        data: data.values,

        backgroundColor: gradient,
        borderWidth: 2,
        pointRadius: 0,
        tension: 0.4,
        fill: true
      }]

          enabled: false
        }

      animation: {
        duration: 2000,
        easing: 'easeOutQuart'
      },
      ...options

    return new Chart(ctx, {
