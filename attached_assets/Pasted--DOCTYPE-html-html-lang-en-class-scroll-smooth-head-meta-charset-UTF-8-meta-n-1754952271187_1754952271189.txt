            priceElements.forEach(element => {
                const currentPrice = parseFloat(element.textContent.replace(/[$,]/g, '').trim());
                const change = (Math.random() - 0.5) * 0.03; // Â±1.5% change
                const newPrice = currentPrice * (1 + change);
                const isPositive = change >= 0;
                
                const parentCard = element.closest('.market-card');
                const changeElement = parentCard.querySelector('[class*="text-sm font-semibold"]');

                if (isPositive) {
                    changeElement.classList.remove('text-red-500');
                    changeElement.classList.add('text-accent');
                    changeElement.textContent = `+${(change * 100).toFixed(2)}%`;
                } else {
                    changeElement.classList.remove('text-accent');
                    changeElement.classList.add('text-red-500');
                    changeElement.textContent = `${(change * 100).toFixed(2)}%`;
                }

                element.textContent = `$${newPrice.toLocaleString('en-US', {
                    minimumFractionDigits: 2,
                    maximumFractionDigits: 2
                })}`;

                // Add a temporary animation to highlight the change
                element.classList.add('transition', 'duration-300', 'ease-in-out', 'transform', 'scale-105');
                setTimeout(() => {
                    element.classList.remove('scale-105');
                }, 300);
            });
        }, 5000);
    </script>
